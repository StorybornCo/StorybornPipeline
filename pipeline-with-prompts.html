1. CLAUDE API — STORY GENERATION
// Claude API — Story Generation
async function generateStory(prompt, apiKey) {
  const model   = document.getElementById('param-model')?.value  || 'claude-sonnet-4-5';
  const maxToks = document.getElementById('param-tokens')?.value || 2000;

  const res = await fetch('https://api.anthropic.com/v1/messages', {
    method: 'POST',
    headers: {
      'Content-Type':      'application/json',
      'x-api-key':         apiKey,
      'anthropic-version': '2023-06-01'
    },
    body: JSON.stringify({
      model:      model,
      max_tokens: parseInt(maxToks),
      messages: [{ role: 'user', content: prompt }]
    })
  });

  if (!res.ok) {
    const e = await res.json();
    throw new Error(e.error?.message || 'Claude API error');
  }
  const data = await res.json();
  return data.content[0].text;
}

// Builds the story prompt from order data
function buildStoryPrompt(o) {
  const pronounMap = {
    'She / Her':   { s:'she',  o:'her',  p:'her',   r:'herself'  },
    'He / Him':    { s:'he',   o:'him',  p:'his',   r:'himself'  },
    'They / Them': { s:'they', o:'them', p:'their', r:'themselves'}
  };
  const pr = pronounMap[o.pronouns] || pronounMap['They / Them'];

  const hairDesc = [o.hairLength, o.hairTexture, o.hairColour, 'hair',
    o.hairStyle ? `(${o.hairStyle})` : ''].filter(Boolean).join(' ');
  const appearanceDesc = `${o.skin}, ${o.eyes}, ${hairDesc}${o.features ? ', ' + o.features : ''}`;

  const supports = [
    o.toy    && `${o.toy} (favourite toy)`,
    o.pet    && `${o.pet} (pet)`,
    o.friend && `${o.friend} (best friend)`
  ].filter(Boolean);

  const world = [
    o.food    && `favourite food: ${o.food}`,
    o.hobbies && `loves: ${o.hobbies}`,
    o.extra
  ].filter(Boolean);

  return `You are a warm, imaginative children's book author.
Write a personalised picture book story. Output ONLY the story.

CHILD: ${o.name}, age ${o.age}. Pronouns: ${pr.s}/${pr.o}/${pr.p}.
Appearance: ${appearanceDesc}. Personality: ${o.personality}.
${world.length ? 'WORLD: ' + world.join(' | ') : ''}
${supports.length ? 'SUPPORTING: ' + supports.join(' | ') : ''}
OCCASION: ${o.occasion}
TONE: ${o.tone}
SETTING: ${o.setting}
SPECIAL MESSAGE: "${o.specialMsg}"

Write exactly 10 pages. Format: "Page N: [2-3 sentence text]"`;
}

// Builds one illustration prompt per page
function buildIllustrationPrompt(o, page) {
  const stylePrefix = 'children\'s picture book illustration, digital_illustration/grain Recraft style, '
    + 'warm pastel tones, gentle ink outlines, cosy storybook aesthetic';
  const hairDesc = [o.hairLength, o.hairTexture, o.hairColour, 'hair'].filter(Boolean).join(' ');
  const charDesc = `a ${o.age}-year-old with ${o.skin}, ${o.eyes}, ${hairDesc}`;
  const scene = page.text.slice(0, 150).replace(/"/g, "'");
  return `${stylePrefix}. Scene: ${scene}. Character: ${charDesc}. `
    + `Setting: ${o.setting}. Mood: warm, magical, joyful.`;
}
